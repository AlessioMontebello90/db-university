QUERY 1:

CONTARE QUANTI ISCRITTI CI SONO STATI OGNI ANNO

SELECT COUNT(*) AS numeroIscritti, YEAR(`enrolment_date`) AS anno
FROM `students`
WHERE YEAR(`enrolment_date`)
GROUP BY YEAR(`enrolment_date`);
